# Double authentification (2FA) TOTP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Activer la 2FA par application d'authentification (TOTP) avec codes de secours : activation depuis Réglages, vérification du code au login, désactivation par mot de passe.

**Architecture:** Hexagonal. Nouveaux ports `Totp` + `Crypto` (adaptateurs otplib / Node crypto), `TokenService` étendu d'un challenge JWT court. Application `two-factor.ts` orchestre activation/confirmation/login/désactivation ; secret TOTP chiffré au repos, codes de secours hachés. UI Réglages + écran code au login.

**Tech Stack:** TypeScript strict, Next.js 16 (routes + client), MongoDB, jose, otplib, qrcode, Vitest.

## Global Constraints

- Secret TOTP **chiffré au repos** (AES-256-GCM), clé **dérivée de `JWT_SECRET`** (`sha256`) — aucune nouvelle variable d'env.
- Codes de secours : **8**, format `XXXX-XXXX`, **usage unique**, stockés **hachés** (sha256), affichés une seule fois.
- `twoFactorSecret`/`twoFactorBackupCodes` **jamais exposés** par `selfUser`/`publicUser` ni dans les logs.
- Challenge de login = JWT **5 min**, claim `scope:"2fa"` ; un challenge ne vaut **jamais** session, une session n'est **jamais** acceptée comme challenge.
- `/api/auth/login/2fa` **rate-limité** comme le login.
- `authenticateUser` **inchangé** (la logique 2FA vit dans la route de login).
- Toute dépendance externe à I/O derrière un port (CLAUDE.md). `qrcode` = encodeur pur (util `lib/qr.ts`).

Créer `src/server/adapters/security/node-crypto.ts` :

```ts
import { createCipheriv, createDecipheriv, createHash, randomBytes } from "node:crypto";
import type { Crypto } from "@/server/ports/security";

/** AES-256-GCM (clé dérivée du secret par sha256) + sha256 pour le hachage de jetons. */
export class NodeCrypto implements Crypto {
  private key: Buffer;
  constructor(secret: string) { this.key = createHash("sha256").update(secret).digest(); }

  encrypt(plain: string): string {
    const iv = randomBytes(12);
    const cipher = createCipheriv("aes-256-gcm", this.key, iv);
    const enc = Buffer.concat([cipher.update(plain, "utf8"), cipher.final()]);
    const tag = cipher.getAuthTag();
    return [iv.toString("base64"), tag.toString("base64"), enc.toString("base64")].join(":");
  }

  decrypt(cipher: string): string {
    const [ivB, tagB, encB] = cipher.split(":");
    if (!ivB || !tagB || !encB) throw new Error("cipher invalide");
    const decipher = createDecipheriv("aes-256-gcm", this.key, Buffer.from(ivB, "base64"));
    decipher.setAuthTag(Buffer.from(tagB, "base64"));
    return Buffer.concat([decipher.update(Buffer.from(encB, "base64")), decipher.final()]).toString("utf8");
  }

  hash(value: string): string { return createHash("sha256").update(value).digest("hex"); }
}
```

```ts
  async verify(token: string) {
    try {
      const { payload } = await jwtVerify(token, this.key);
      if (payload.scope === "2fa") return null; // un challenge n'est pas une session
      return typeof payload.userId === "string" ? { userId: payload.userId } : null;
    } catch { return null; }
  }
  async issueChallenge(userId: string) {
    return new SignJWT({ userId, scope: "2fa" })
      .setProtectedHeader({ alg: "HS256" })
      .setIssuedAt()
      .setExpirationTime("5m")
      .sign(this.key);
  }
  async verifyChallenge(token: string) {
    try {
      const { payload } = await jwtVerify(token, this.key);
      return payload.scope === "2fa" && typeof payload.userId === "string" ? { userId: payload.userId } : null;
    } catch { return null; }
  }
```

- [ ] **Step 4: Implement the application module**

Créer `src/server/application/two-factor.ts` :

```ts
import type { Deps } from "@/server/container";
import { AuthError, NotFoundError, ValidationError } from "@/server/domain/errors";
import { qrSvgFromUri } from "@/lib/qr";

const ISSUER = "vulu";
const BACKUP_CODE_COUNT = 8;
const BACKUP_ALPHABET = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789"; // sans I,O,0,1 (lisibilité)

function randomBackupCode(): string {
  const pick = (n: number) => Array.from({ length: n }, () => BACKUP_ALPHABET[Math.floor(Math.random() * BACKUP_ALPHABET.length)]).join("");
  return `${pick(4)}-${pick(4)}`;
}

function normalizeCode(code: string): string {
  return code.replace(/[\s-]/g, "").toUpperCase();
}

async function loadUser(deps: Deps, userId: string) {
  const user = await deps.users.findById(userId);
  if (!user) throw new NotFoundError("Utilisateur introuvable");
  return user;
}

export async function beginTwoFactorSetup(deps: Deps, userId: string): Promise<{ keyUri: string; secret: string; qrSvg: string }> {
  const user = await loadUser(deps, userId);
  const secret = deps.totp.generateSecret();
  const keyUri = deps.totp.keyUri(secret, user.username, ISSUER);
  await deps.users.update({ ...user, twoFactorSecret: deps.crypto.encrypt(secret) });
  return { keyUri, secret, qrSvg: await qrSvgFromUri(keyUri) };
}

export async function confirmTwoFactorSetup(deps: Deps, userId: string, code: string): Promise<{ backupCodes: string[] }> {
  const user = await loadUser(deps, userId);
  if (!user.twoFactorSecret) throw new ValidationError("Aucune configuration 2FA en attente");
  const secret = deps.crypto.decrypt(user.twoFactorSecret);
  if (!deps.totp.verify(normalizeCode(code), secret)) throw new AuthError("Code invalide");
  const backupCodes = Array.from({ length: BACKUP_CODE_COUNT }, randomBackupCode);
  await deps.users.update({
    ...user,
    twoFactorEnabled: true,
    twoFactorBackupCodes: backupCodes.map((c) => deps.crypto.hash(normalizeCode(c))),
  });
  return { backupCodes };
}

export async function disableTwoFactor(deps: Deps, userId: string, password: string): Promise<void> {
  const user = await loadUser(deps, userId);
  if (!(await deps.hasher.verify(password, user.passwordHash))) throw new AuthError("Mot de passe invalide");
  await deps.users.update({ ...user, twoFactorEnabled: false, twoFactorSecret: null, twoFactorBackupCodes: [] });
}

export async function completeTwoFactorLogin(deps: Deps, challenge: string, code: string): Promise<{ token: string }> {
  const payload = await deps.tokens.verifyChallenge(challenge);
  if (!payload) throw new AuthError("Session de vérification expirée");
  const user = await loadUser(deps, payload.userId);
  if (!user.twoFactorEnabled || !user.twoFactorSecret) throw new AuthError("2FA non configurée");

  const normalized = normalizeCode(code);
  const totpOk = deps.totp.verify(normalized, deps.crypto.decrypt(user.twoFactorSecret));
  if (!totpOk) {
    const hash = deps.crypto.hash(normalized);
    if (!user.twoFactorBackupCodes.includes(hash)) throw new AuthError("Code invalide");
    await deps.users.update({ ...user, twoFactorBackupCodes: user.twoFactorBackupCodes.filter((h) => h !== hash) });
  }
  return { token: await deps.tokens.issue({ userId: user.id }) };
}
```

```tsx
"use client";
import { useState } from "react";
import { api } from "@/lib/api-client";
import { toast } from "@/lib/toast";

type Step = "idle" | "setup" | "codes";

export function TwoFactorSettings({ enabled: initialEnabled }: { enabled: boolean }) {
  const [enabled, setEnabled] = useState(initialEnabled);
  const [step, setStep] = useState<Step>("idle");
  const [qrSvg, setQrSvg] = useState("");
  const [secret, setSecret] = useState("");
  const [code, setCode] = useState("");
  const [backupCodes, setBackupCodes] = useState<string[]>([]);
  const [password, setPassword] = useState("");
  const [busy, setBusy] = useState(false);

  async function startSetup() {
    setBusy(true);
    try {
      const d = await api.post<{ keyUri: string; secret: string; qrSvg: string }>("/api/me/2fa/setup");
      setQrSvg(d.qrSvg); setSecret(d.secret); setStep("setup");
    } catch { toast("Impossible de démarrer la configuration", "error"); } finally { setBusy(false); }
  }

  async function confirm() {
    setBusy(true);
    try {
      const d = await api.post<{ backupCodes: string[] }>("/api/me/2fa/confirm", { code });
      setBackupCodes(d.backupCodes); setStep("codes"); setEnabled(true); setCode("");
    } catch { toast("Code invalide, réessaie", "error"); } finally { setBusy(false); }
  }

  async function disable() {
    setBusy(true);
    try {
      await api.post("/api/me/2fa/disable", { password });
      setEnabled(false); setPassword(""); toast("2FA désactivée");
    } catch { toast("Mot de passe invalide", "error"); } finally { setBusy(false); }
  }

  function downloadCodes() {
    const blob = new Blob([backupCodes.join("\n") + "\n"], { type: "text/plain" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url; a.download = "vulu-codes-de-secours.txt"; a.click();
    URL.revokeObjectURL(url);
  }

  if (step === "codes") {
    return (
      <div className="rounded-2xl border border-[var(--color-border)] p-4">
        <p className="text-sm font-medium">Codes de secours</p>
        <p className="mt-1 text-xs text-[var(--color-text-muted)]">Conserve-les en lieu sûr. Chaque code ne sert qu'une fois. Ils ne seront plus affichés.</p>
        <ul className="my-3 grid grid-cols-2 gap-1 font-mono text-sm">
          {backupCodes.map((c) => <li key={c}>{c}</li>)}
        </ul>
        <div className="flex gap-2">
          <button onClick={downloadCodes} className="rounded-full bg-[var(--color-primary)] px-4 py-2 text-sm font-semibold text-white">Télécharger</button>
          <button onClick={() => { navigator.clipboard.writeText(backupCodes.join("\n")); toast("Copié"); }} className="rounded-full border border-[var(--color-border)] px-4 py-2 text-sm font-semibold">Copier</button>
          <button onClick={() => setStep("idle")} className="ml-auto rounded-full px-4 py-2 text-sm font-semibold text-[var(--color-text-muted)]">Terminé</button>
        </div>
      </div>
    );
  }

  if (step === "setup") {
    return (
      <div className="rounded-2xl border border-[var(--color-border)] p-4">
        <p className="text-sm font-medium">Scanne ce QR dans ton application d'authentification</p>
        <div className="my-3 w-40" dangerouslySetInnerHTML={{ __html: qrSvg }} />
        <p className="text-xs text-[var(--color-text-muted)]">Ou saisis cette clé : <span className="font-mono">{secret}</span></p>
        <input value={code} onChange={(e) => setCode(e.target.value)} inputMode="numeric" placeholder="Code à 6 chiffres"
          className="mt-3 w-full rounded-full border border-[var(--color-border)] bg-[var(--color-surface)] px-4 py-2.5 text-sm" />
        <div className="mt-3 flex gap-2">

          <button onClick={() => setStep("idle")} className="rounded-full px-4 py-2 text-sm font-semibold text-[var(--color-text-muted)]">Annuler</button>
        </div>
      </div>
    );
  }

  return (
    <div className="flex items-center justify-between gap-3">
      <div>
        <p className="text-sm font-medium">Double authentification (2FA)</p>
        <p className="text-xs text-[var(--color-text-muted)]">Application d'authentification (TOTP).</p>
      </div>
      {enabled ? (
        <div className="flex items-center gap-2">
          <input type="password" value={password} onChange={(e) => setPassword(e.target.value)} placeholder="Mot de passe"
            className="w-32 rounded-full border border-[var(--color-border)] bg-[var(--color-surface)] px-3 py-1.5 text-sm" />
          <button disabled={busy} onClick={disable} className="rounded-full border border-[var(--color-border)] px-3 py-1.5 text-xs font-semibold disabled:opacity-50">Désactiver</button>
        </div>
      ) : (
        <button disabled={busy} onClick={startSetup} className="rounded-full bg-[var(--color-primary)] px-4 py-1.5 text-xs font-semibold text-white disabled:opacity-50">Activer</button>
      )}
    </div>
  );
}
```
